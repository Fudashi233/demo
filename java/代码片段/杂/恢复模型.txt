package cn.edu.jxau;
public class Test
{
	public static void main(String[] args)
	{
		int count = -2;
		while(true)
		{
			try
			{
				if(count<0)
					throw new NotNegativeException();
			}
			catch(NotNegativeException ex)
			{
				count++;
				if(count>=0)
					break;
			}
		}
		System.out.println(count);
	}
}
class NotNegativeException extends Exception
{
	
}
