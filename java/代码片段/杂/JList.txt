import javax.swing.JApplet;
import javax.swing.JLabel;
import javax.swing.JComboBox;
import javax.swing.JSpinner;
import javax.swing.SpinnerNumberModel;
import javax.swing.JList;
import javax.swing.JPanel;
import javax.swing.JScrollPane;
import javax.swing.event.ChangeEvent;
import javax.swing.event.ChangeListener;
import javax.swing.event.ListSelectionEvent;
import javax.swing.event.ListSelectionListener;

import java.awt.GridLayout;
import java.awt.BorderLayout;
import java.awt.event.ActionListener;
import java.awt.event.ActionEvent;
public class Applet extends JApplet
{
	public JSpinner visibleRowCount;
	public JComboBox<String> layoutOrientation;
	public JComboBox<String> selectionMode;
	public JList<String> items;
	public JLabel display;
	public Applet()
	{
		visibleRowCount=new JSpinner(new SpinnerNumberModel(2,1,3,1));
		String[] orientation={"VERTiCAL","VERTICAL_WRAP","HORIZONTAL_WRAP"};
	    layoutOrientation=new JComboBox<String>(orientation);
		String[] mode={"SINGLE_SELECTION","SINGLE_INTERVAL_SELECTION","MULTIPLE_INTERVAL_SELECTION"};
		selectionMode=new JComboBox<String>(mode);
		selectionMode.setSelectedIndex(2);
		JPanel p1=new JPanel(new GridLayout(3,2,0,0));
		p1.add(new JLabel("VisibleRowCount"));
		p1.add(visibleRowCount);
		p1.add(new JLabel("layoutOrientation"));
		p1.add(layoutOrientation);
		p1.add(new JLabel("selectionMode"));
		p1.add(selectionMode);
		
	    String[] item={"item1","item2","item3","item4","item5","item6","item7"};
		items=new JList<String>(item);
		
		display=new JLabel("");
		JPanel p2=new JPanel();
		p2.add(display);
		
		this.add(p1,BorderLayout.NORTH);
		this.add(new JScrollPane(items));
		this.add(p2,BorderLayout.SOUTH);
		visibleRowCount.addChangeListener(
		                                        new ChangeListener()
												{
													public void stateChanged(ChangeEvent e)
													{
														int temp=((Integer)(visibleRowCount.getValue())).intValue();
														items.setVisibleRowCount(temp);
													}
												}
		                                 );
	    layoutOrientation.addActionListener(
		                                        new ActionListener()
												{
													public void actionPerformed(ActionEvent e)
													{
														items.setLayoutOrientation(layoutOrientation.getSelectedIndex());
													}
												}
		                                   );
		selectionMode.addActionListener(
		                                    new ActionListener()
											{
												public void actionPerformed(ActionEvent e)
												{
													items.setSelectionMode(selectionMode.getSelectedIndex());
												}
											}
		                               );
		items.addListSelectionListener(
		                                    new ListSelectionListener()
											{
												public void valueChanged(ListSelectionEvent e)
												{
													Object[] temp=items.getSelectedValues();
													String text="";
													for(int i=0;i<temp.length;i++)
														System.out.print(temp[i]);
													for(int i=0;i<temp.length;i++)
														text=text+temp[i].toString()+"  ";
													display.setText(text);
												}
											}
		                              );
		
	}
}
