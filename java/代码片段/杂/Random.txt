import java.util.Random;
public class a
{
	public static void main(String[] args)
	{
		Random r=new Random();//无参数  表示已当前时间为种子
		int[] a=new int[10];
		for(int i=0;i<a.length;i++)
			a[i]=r.nextInt(10)+1;
		for(int u:a)
			System.out.print(u+"\t");
		
	//	nextInt() nextLong()   nextDouble() nextFloat()  nextBoolean()
 	}
}
