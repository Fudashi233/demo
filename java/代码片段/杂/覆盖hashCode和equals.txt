class Demo
{
	public String st;
	public int id;
	public Demo(String st,int id)
	{
		this.st = st;
		this.id = id;
	}
	@Override()
	public int hashCode()
	{
		final int HASH = 37;
		int result = 17;
		result = HASH*result + st.hashCode();
		result = HASH*result + id;
		return result;
	}
	public boolean equals(Object o)
	{
		return o instanceof Demo 
		       && ((Demo)o).id==this.id
			   && this.st.equals(((Demo)o).st);
	}
}
