public class Test1
{
	public static void main(String[] agrs)
	{
		int a = 90;
		System.out.print(a+"=");
		f(a);
	}
	private static void f(int a)
	{
		int i=2;
		for( ;i<a/2;i++)
		{
			if(a%i==0)
			{
				a/=i;
				System.out.print(i+"*");
				i--;
			}
		}
		System.out.println(a);
	}
}
