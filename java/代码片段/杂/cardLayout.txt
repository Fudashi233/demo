import javax.swing.JButton;
import javax.swing.JComboBox;
import javax.swing.ImageIcon;
import javax.swing.JPanel;
import javax.swing.JApplet;
import javax.swing.JLabel;

import java.awt.CardLayout;
import java.awt.FlowLayout;
import java.awt.BorderLayout;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.awt.event.ItemEvent;
import java.awt.event.ItemListener;
public class MyApplet extends JApplet
{
    private static final int NUM_OF_ICON=4;
    private ImageIcon[] icon=new ImageIcon[4];	
	private JButton first;
	private JButton last;
	private JButton next;
	private JButton previous;
	private JComboBox jcbo;
	private CardLayout cardLayout;
	public MyApplet()
	{
		icon[0]=new ImageIcon("p1.jpg");
		JLabel l1=new JLabel(icon[0]);
		icon[1]=new ImageIcon("p2.jpg");
		JLabel l2=new JLabel(icon[1]);
		icon[2]=new ImageIcon("p3.jpg");
		JLabel l3=new JLabel(icon[2]);
		icon[3]=new ImageIcon("p4.jpg");
		JLabel l4=new JLabel(icon[3]);
		cardLayout=new CardLayout(10,20);
		JPanel p1=new JPanel(cardLayout);
		p1.add(l1,"1");
		p1.add(l2,"2");
		p1.add(l3,"3");
		p1.add(l4,"4");
		
		first=new JButton("First");
		last=new JButton("Last");
		next=new JButton("next");
		previous=new JButton("Previous");
		JLabel image=new JLabel("Image");
		jcbo=new JComboBox();
		jcbo.addItem("1");
		jcbo.addItem("2");
		jcbo.addItem("3");
		jcbo.addItem("4");
		JPanel p2=new JPanel();
		p2.add(first);
		p2.add(last);
		p2.add(next);
		p2.add(previous);
		p2.add(image);
		p2.add(jcbo);
		
		this.setLayout(new BorderLayout(5,5));
		this.add(p1,BorderLayout.CENTER);
		this.add(p2,BorderLayout.SOUTH);
		//fit listener
		first.addActionListener(
		                            new ActionListener()
									{
										public void actionPerformed(ActionEvent e)
										{
											cardLayout.first(p1);
											
										}
									}
		                       );
		last.addActionListener(
		                            new ActionListener()
									{
										public void actionPerformed(ActionEvent e)
										{
											cardLayout.last(p1);
										}
									}
		                      );
		next.addActionListener(
		                            new ActionListener()
									{
										public void actionPerformed(ActionEvent e)
										{
											cardLayout.next(p1);
										}
									}
		                      );
	    previous.addActionListener(
		                                new ActionListener()
										{
											public void actionPerformed(ActionEvent e)
											{
												cardLayout.previous(p1);
											}
										}
		                          );
		jcbo.addItemListener(
		                            new ItemListener()
									{
										public void itemStateChanged(ItemEvent e)
										{
											cardLayout.show(p1,(String)e.getItem());
										}
									}
		                    );
		
	}
}
