import javax.swing.*;
import java.awt.Graphics;
import java.awt.Font;
import java.awt.FontMetrics;
import java.awt.Color;
import java.awt.FlowLayout;
import java.awt.BorderLayout;
import java.awt.GridLayout;
import java.awt.Dimension;
public class a
{
	public static void main(String[] args)
	{
		MyFrame frame=new MyFrame();
		frame.setSize(500,450);
		frame.setLocationRelativeTo(null);
		frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
		frame.setVisible(true);
	}
}


class MyPanel extends JPanel
{
	private String message_;
	public MyPanel()
	{
		message_="java";
	}
	public MyPanel(String m)
	{
		message_=m;
	}
	protected void paintComponent(Graphics g)
	{
		super.paintComponent(g);
		Font f=new Font("Californian FB",Font.BOLD,20);
		FontMetrics fm=g.getFontMetrics();
		g.setFont(f);
		g.drawString(message_,getWidth()/2-fm.stringWidth(message_),getHeight()/2-fm.getAscent()/2);
	}
	public Dimension getPreferredSize()
	{
		return new Dimension(80,80);
	}
}


class MyFrame extends JFrame
{
	public MyFrame()
	{
		super("Windows");
		setLayout(new FlowLayout(FlowLayout.LEFT,0,0));
		JPanel p1=new JPanel(new GridLayout(3,1,0,0));
		JPanel p2=new JPanel(new GridLayout(1,1,0,0));
		MyPanel a1=new MyPanel("C++");
		MyPanel a2=new MyPanel("HTML");
		MyPanel a3=new MyPanel("PHP");
		MyPanel a4=new MyPanel("Java");
		a4.setBackground(Color.BLACK);
		a1.setBackground(Color.RED);
		a2.setBackground(Color.YELLOW);
		a3.setBackground(Color.RED);
		p1.add(a1);
		p1.add(a2);
		p1.add(a3);
		p2.add(a4);
		//p2.setPreferredSize(new Dimension(200,450));
		add(p2);
		//p1.setPreferredSize(new Dimension(250,450));
		add(p1);
	}
}
