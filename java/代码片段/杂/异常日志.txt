package cn.edu.jxau;

import java.io.IOException;
import java.io.StringWriter;
import java.io.PrintWriter;
import java.util.logging.Logger;
public class Test
{
	public static void main(String[] args) 
	{
		try
		{
		    g();	
		}
		catch(NotNegativeException ex)
		{
			ex.printStackTrace();
		}
			
	}
	public static void f() throws NotNegativeException 
	{
		throw new NotNegativeException();
	}
	public static void g() throws NotNegativeException 
	{
		f();
	}
}
class NotNegativeException extends Exception
{
	private Logger logger = Logger.getLogger("NotNegativeException");
	public NotNegativeException()
	{
		StringWriter trace = new StringWriter();
		printStackTrace(new PrintWriter(trace));
		logger.severe(trace.toString());
		logger.warning(trace.toString());
	}
}
